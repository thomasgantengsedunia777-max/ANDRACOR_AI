<script>
function ai(input){
    input = input.toLowerCase();

    if(input.includes("halo") || input.includes("hai"))
        return "Halo Tuan 🧑‍💻🧑‍💻";

    if(input.includes("siapa kamu"))
        return "Saya AI buatan Tuan 😎";

    if(input.includes("apa kabar"))
        return "Saya selalu siap melayani Tuan";

    if(input.includes("jam")){
        let now = new Date();
        return "Sekarang jam: " + now.toLocaleTimeString();
    }

    if(input.includes("+") || input.includes("-") || input.includes("*") || input.includes("/")){
        try {
            return "Hasil: " + eval(input);
        } catch {
            return "Saya tidak mengerti";
        }
    }

    let random = [
        "Menarik 🤔",
        "Coba jelasin lagi",
        "Saya masih belajar",
        "Oke Tuan"
    ];

    return random[Math.floor(Math.random() * random.length)];
}

function kirim(){
    let input = document.getElementById("input").value;
    let chat = document.getElementById("chat");

    if(input === "") return;

    chat.innerHTML += "<p><b>Tuan:</b> " + input + "</p>";
    chat.innerHTML += "<p><b>AI:</b> " + ai(input) + "</p>";

    document.getElementById("input").value = "";
    chat.scrollTop = chat.scrollHeight;
}
